현재 시각 :

innertext = 텍스트표시
innerHtml = html 태그 반영하여 텍스트 표시 

            function innText(){
                let now = new Date();
                document.getElementById("current").innerText = "<em>" + now + "</em>";
            }
            
            function innHtml(){
                let now = new Date();
                document.getElementById("current").innerHTML = "<em>" + now + "</em>";
            }
