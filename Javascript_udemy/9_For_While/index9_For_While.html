반복문

for

정확한 범위를 알때 사용

                for(let i =0; i<10; i++){
                    console.log("for",i);
                }
            
    짝수만 나오게
                
  • if 사용
                for(let i =0; i<10; i++){
                    if(i%2==0){
                        console.log("for 짝수 : ",i);
                    }
                }
    • 
                
  • 증감식 수정
                for(let i =0; i<10; i+=2){    
                    console.log("for 짝수 : ",i);
                }
    •
for문, 배열과 함께 자주 사용

while

정확한 범위를 알 수 없을때

                let a=1;
                while(a>10){
                    console.log("while문 실행"+ a);
                    a++;
                }
                버튼이 클릭되었을때 실행
                while(buttonclick == true){
                    console.log("while문 실행"+ a);
                }
