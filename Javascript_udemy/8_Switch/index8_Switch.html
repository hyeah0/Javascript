switch

값이 하나로 딱 떨어지는 경우에만 사용이 가능 

                switch (menu){
                    case 1 : 
                        console.log("파스타");
                        break;
                    case 2 : 
                        console.log("치킨");
                        break;
                    case 3 : 
                        console.log("피자");
                        break;
                    default :
                        console.log("메뉴를 골라주세요");
                }

                2,3일때 값을 정할때
                case 2 :
                case 3 :
                    console.log("");

삼항연산식

리턴할 값이 1개 일때

                if(menu <=3){
                    console.log("범위안에 숫자입니다.");
                }
                else{
                    console.log("범위밖에 숫자입니다.");
                }

                삼항연산식으로 표현
                let answer = menu <=3? "범위안에 숫자입니다.":"범위밖의 숫자입니다.";
                console.log(answer);
