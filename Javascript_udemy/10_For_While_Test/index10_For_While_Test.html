                    if(stringValue[b] == "3" || stringValue[b] == "6" || stringValue[b] == "9"){

                        console.log("stringValue[1] : "+stringValue[1]);
                        stringValue.length[0] = a의 문자의 0번째 문자
                        stringValue.length[1] = a의 문자의 1번째 문자
                        예시)36의 0번째 문자는 3
                             , 1번째 문자는 6 이다.

                        console.log("result : " + result);
                    }
                
                }
                console.log("a : "+ a);
                console.log("result.length : " + result.length)
                result문자의 길이
                
                console.log(result.length>0?result:a)
                result문자의 길이가 0보다 클때 result 값을 반환, 아닐경우 a값을 반환한다
