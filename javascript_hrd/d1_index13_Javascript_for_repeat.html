반복문

반복문을 사용하지 않는 문

            var sum = 0;
            sum += 1;
            sum += 2;
            sum += 3;
            sum += 4;
            sum += 5;
            document.write("1부터 5까지의 합은" , sum , "입니다.");

반복문(for)를 사용

            var i;
            var add = 0;

            for(i=0; i<6; i++ ){
                    add += i;
            }
            document.write("1부터 5까지의 합은" , add , "입니다".); 
        
            // 구구단
            var a;
            var b;

            for(a = 2; a <= 9; a++){
                document.write("<div>")
                    
                    document.write("<h5>" , a , "단<h5>");
                    for(b = 1; b <= 9; b++){
                        document.write(a , "X" , b , "=" , a*b , "<br>");
                    }
                    
                document.write("</div>")
                }
