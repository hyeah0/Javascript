            var coffee = prompt("밀크,블랙,카푸치노 중 하나를 적어주세요","");


            switch(coffee){
                case "밀크" : alert("밀크 커피 입니다.");
                        break
                case "블랙" : alert("블랙 커피 입니다.");
                        break
                case "카푸치노" : alert("카푸치노 입니다.");
                        break
                default : alert("선택하신 커피종류가 없습니다.");
            }
